function updateSignal(sym) { const d = data[sym]; const m = marketData[sym] || {}; // === 1. CALCULATE INDIVIDUAL SCORES === let scores = { cvd: 0, vwap: 0, rsi: 0, volume: 0, funding: 0, momentum: 0 }; // CVD Score (-2 to +2) if (Math.abs(d.cvd) > 50) scores.cvd = d.cvd > 0 ? 2 : -2; else if (Math.abs(d.cvd) > 20) scores.cvd = d.cvd > 0 ? 1.5 : -1.5; else if (Math.abs(d.cvd) > 5) scores.cvd = d.cvd > 0 ? 1 : -1; else scores.cvd = d.cvd > 0 ? 0.3 : -0.3; // VWAP Score (-1.5 to +1.5) const vwapDist = (d.price - d.vwap) / d.vwap * 100; if (Math.abs(vwapDist) > 0.3) scores.vwap = d.price > d.vwap ? 1.5 : -1.5; else if (Math.abs(vwapDist) > 0.1) scores.vwap = d.price > d.vwap ? 1 : -1; else scores.vwap = d.price > d.vwap ? 0.5 : -0.5; // RSI Score (-1.5 to +1.5) if (d.rsi < 25) scores.rsi = 1.5; // Strong oversold else if (d.rsi < 35) scores.rsi = 1; // Oversold else if (d.rsi < 45) scores.rsi = 0.5; // Slightly bullish else if (d.rsi < 55) scores.rsi = 0; // Neutral else if (d.rsi < 65) scores.rsi = -0.5; // Slightly bearish else if (d.rsi < 75) scores.rsi = -1; // Overbought else scores.rsi = -1.5; // Strong overbought // Volume Score (-2 to +2) - THIS IS THE KEY if (d.volRatio > 2.5) scores.volume = 2; // Very high volume else if (d.volRatio > 1.8) scores.volume = 1.5; else if (d.volRatio > 1.2) scores.volume = 1; else if (d.volRatio > 0.8) scores.volume = 0.5; else if (d.volRatio > 0.5) scores.volume = -0.5; else scores.volume = -1.5; // Dangerously low // Funding Rate Score (-1 to +1) if (m.fundingRate > 0.02) scores.funding = -1; else if (m.fundingRate > 0.01) scores.funding = -0.5; else if (m.fundingRate < -0.01) scores.funding = 1; else if (m.fundingRate < -0.005) scores.funding = 0.5; else scores.funding = 0; // Momentum Score (CVD change) (-1.5 to +1.5) if (d.cvdHistory && d.cvdHistory.length > 0) { const cvdChange = d.cvd - d.cvdHistory[d.cvdHistory.length - 1]; if (cvdChange > 10) scores.momentum = 1.5; else if (cvdChange > 5) scores.momentum = 1; else if (cvdChange > 2) scores.momentum = 0.5; else if (cvdChange < -10) scores.momentum = -1.5; else if (cvdChange < -5) scores.momentum = -1; else if (cvdChange < -2) scores.momentum = -0.5; else scores.momentum = 0; } // Store CVD history if (!d.cvdHistory) d.cvdHistory = []; d.cvdHistory.push(d.cvd); if (d.cvdHistory.length > 3) d.cvdHistory.shift(); // === 2. CALCULATE VOLUME MULTIPLIER === // Volume adjusts the THRESHOLD, not the signal itself let volumeMultiplier = 1.0; let confidenceBoost = 0; if (d.volRatio > 2.0) { volumeMultiplier = 0.6; // Easier to trigger (high conviction) confidenceBoost = 20; } else if (d.volRatio > 1.5) { volumeMultiplier = 0.75; confidenceBoost = 10; } else if (d.volRatio > 1.2) { volumeMultiplier = 0.9; confidenceBoost = 5; } else if (d.volRatio > 0.8) { volumeMultiplier = 1.1; // Harder to trigger (low conviction) confidenceBoost = -10; } else if (d.volRatio > 0.5) { volumeMultiplier = 1.4; // Much harder (very low volume) confidenceBoost = -25; } else { volumeMultiplier = 1.8; // Extremely hard (danger zone) confidenceBoost = -40; } // === 3. CALCULATE RAW & WEIGHTED SCORE === const rawScore = scores.cvd * 2.0 + // CVD is most important scores.vwap * 1.5 + // VWAP trend scores.rsi * 1.0 + // RSI momentum scores.volume * 2.5 + // Volume is CRITICAL scores.funding * 0.8 + // Funding rate scores.momentum * 1.2; // CVD acceleration // Apply volume multiplier to threshold const weightedScore = rawScore; // === 4. DYNAMIC THRESHOLDS === const buyThreshold = 4.0 * volumeMultiplier; const sellThreshold = -4.0 * volumeMultiplier; const strongBuyThreshold = 7.0 * volumeMultiplier; const strongSellThreshold = -7.0 * volumeMultiplier; // === 5. GENERATE SIGNAL === let signal, signalClass; if (weightedScore >= strongBuyThreshold) { signal = 'STRONG BUY 🔥'; signalClass = 'strong-buy'; } else if (weightedScore >= buyThreshold) { signal = 'BUY'; signalClass = 'buy'; } else if (weightedScore <= strongSellThreshold) { signal = 'STRONG SELL 🔥'; signalClass = 'strong-sell'; } else if (weightedScore <= sellThreshold) { signal = 'SELL'; signalClass = 'sell'; } else { signal = 'NEUTRAL ⏸️'; signalClass = 'neutral'; } // === 6. CALCULATE CONFIDENCE === const maxScore = 12; // Theoretical max const baseConfidence = (Math.abs(weightedScore) / maxScore) * 100; d.confidence = Math.max(0, Math.min(100, baseConfidence + confidenceBoost)); // === 7. STORE RESULTS === d.score = weightedScore; d.signal = signal; d.signalClass = signalClass; d.indicatorScores = scores; d.volumeMultiplier = volumeMultiplier; }
